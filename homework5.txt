create table products1(product_id NUMBER, product_name VARCHAR2(50), price NUMBER,category VARCHAR2(50));

--6
CREATE OR REPLACE PROCEDURE add_product (p_product_id   IN NUMBER, p_product_name IN VARCHAR2, p_price IN NUMBER, p_category IN VARCHAR2)
AS
BEGIN
    INSERT INTO products1
    VALUES (p_product_id, p_product_name, p_price, p_category);
    DBMS_OUTPUT.PUT_LINE('Product added');
END;
/

--7
BEGIN
    add_product(101, 'Laptop', 800, 'Electronics');
    add_product(102, 'Chair', 50, 'Furniture');
END;
/

--8
CREATE OR REPLACE PROCEDURE update_product (p_product_id IN NUMBER, p_price IN NUMBER, p_category IN VARCHAR2)
AS
BEGIN
    UPDATE products1
    SET price = p_price,
        category = p_category
    WHERE product_id = p_product_id;
    DBMS_OUTPUT.PUT_LINE('Product updated');
END;
/

--9
BEGIN
    update_product(101, 750, 'Office Electronics');
END;
/

--10
CREATE OR REPLACE PROCEDURE delete_product1 (p_product_id IN NUMBER)
AS
BEGIN
    DELETE FROM products1
    WHERE product_id = p_product_id;
    DBMS_OUTPUT.PUT_LINE('Product deleted');
END;
/
