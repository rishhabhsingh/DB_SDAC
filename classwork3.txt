--CW02
SELECT AVG(Salary) AS AverageSalary
FROM Employees;



--CW03
SELECT MAX(Salary) AS MaxSalary
FROM Employees;



--CW04

FROM Employees;



--CW05
SELECT UPPER(FirstName) AS FirstName_Upper,
       UPPER(LastName) AS LastName_Upper
FROM Employees;



--CW06
SELECT FirstName, LENGTH(FirstName) AS FirstName_Length
FROM Employees;



--CW07
SELECT CONCAT(FirstName, ' ', LastName) AS FullName
FROM Employees;



--CW08
SELECT * FROM Employees ORDER BY salary DESC FETCH FIRST 5 ROWS ONLY;



--CW09
SELECT department_id, SUM(salary) AS total_salary FROM Employees GROUP BY department_id;



--CW10
SELECT department_id, AVG(salary) AS avg_salary FROM Employees GROUP BY department_id ORDER BY avg_salary DESC FETCH FIRST 1 ROW ONLY;



--CW11
SELECT department_id, COUNT(*) AS employee_count FROM Employees GROUP BY department_id HAVING COUNT(*) > 10;



--CW12
SELECT department_id, MIN(salary) AS min_salary, MAX(salary) AS max_salary FROM Employees GROUP BY department_id;



--CW13
